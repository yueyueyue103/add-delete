test("测试", async () => {
  const ev = eventJson;
  ev.body = JSON.stringify({
    预约番号: "123456",
    履历番号: "3",
    更新日期: "2023/11",
    定期卷情报确认: "", // 为空
  });

  // 连接数据库
  const client = await pool.connect();

  try {
    // 执行数据库操作
    await client.query('更新回原来的日期操作');

    // 调用handler函数，并获取响应
    const resp = await handler(ev.context);

    // 添加您的其他断言和期望
    expect(resp.statusCode).toBe(200);
    // 假设handler函数返回一个JSON字符串，您可以使用JSON.parse()来解析响应
    const responseBody = JSON.parse(resp.body);
    // 然后验证预期的结果
    expect(responseBody.更新日时).toBeUndefined();
    expect(responseBody.更新件数).toBe(1);
  } finally {
    // 在操作完成后释放连接回连接池
    client.release();
  }
});

import React, { useState } from 'react';
import './c2.css'; // 引入CSS文件


const TextArea = () => {
  const [selectedOption, setSelectedOption] = useState('');
  const [userInput, setUserInput] = useState('');

  const handleSelectChange = (e) => {
    const selectedOption = e.target.value;
    setSelectedOption(selectedOption);

    // 根据选项设置对应的文本内容
    const textForOption = getTextForOption(selectedOption);
    setUserInput(textForOption);
  };

  const handleTextAreaChange = (e) => {
    setUserInput(e.target.value);
  };

  const handleSubmit = () => {
    // 在此处添加提交逻辑，可以将 userInput 的内容提交到服务器或执行其他操作
    console.log('提交内容:', userInput);
  };

  const getTextForOption = (option) => {
    // 根据选项返回对应的文本内容
    switch (option) {
      case 'option1':
        return '这是选项1的内容';
      case 'option2':
        return '这是选项2的内容';
      case 'option3':
        return '这是选项3的内容';
      case 'option4':
        return '这是选项4的内容';
      default:
        return '';
    }
  };

  return (
    <div>
      <select className="container" value={selectedOption} onChange={handleSelectChange}>
        <option value="">请选择</option>
        <option value="option1">选项1</option>
        <option value="option2">选项2</option>
        <option value="option3">选项3</option>
        <option value="option4">选项4</option>
      </select>

      <textarea
        className="textarea"
        value={userInput}
        onChange={handleTextAreaChange}
        placeholder="在此输入内容"
      />
      <br></br><br></br><br></br><br></br>
      <button onClick={handleSubmit}>提交</button>
    </div>
  );
};

export default TextArea;
